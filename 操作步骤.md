REM 0. 安装 Miniconda download

REM 1. 进入项目目录
cd E:\work\adv_attack

REM 2. 接受 Anaconda 官方源的 ToS（只需做一次）
conda tos accept --override-channels --channel https://repo.anaconda.com/pkgs/main
conda tos accept --override-channels --channel https://repo.anaconda.com/pkgs/r
conda tos accept --override-channels --channel https://repo.anaconda.com/pkgs/msys2

REM 3. 创建并激活 anyattack 环境（Python 3.8）
cocreate -n anyattack python=3.8 -y
conda activate anyattacknda 

REM 4. 安装 PyTorch（CPU 版；如果有 GPU 再按官网改成对应 CUDA 版本）
pip install torch torchvision

REM 5. 安装 LAVIS
用 conda-forge 安装一个兼容 Python 3.8 的 spaCy 版本
conda install -c conda-forge spacy=3.6.1 -y

再装 LAVIS，此时 pip 不会再去编译最新的 spaCy
pip install salesforce-lavis

REM 6. 安装本项目常用依赖
pip install numpy pandas pyarrow pillow matplotlib requests tqdm webdataset omegaconf timm ftfy regex packaging scikit-learn

REM 7. 安装 COCO 评测相关库
pip install pycocotools-windows pycocoevalcap

REM 7.1 安装 UI 依赖（Streamlit）
pip install -r ui\requirements-ui.txt

REM 8. 设置数据集和项目路径（按你实际数据集路径修改 D:\datasets）
set DATASET_BASE_PATH=E:\work\adv_attack\datasets
set PROJECT_PATH=E:\work\adv_attack

REM 9.（手动操作，非命令）从 README 里的 OneDrive 链接下载 checkpoints 和 json，
REM    解压后确保目录结构类似：
REM      E:\work\adv_attack\checkpoints\coco_cos.pt 等
REM      E:\work\adv_attack\json\coco_caption_target.json 等

REM 10. 测试 demo.py 是否能跑通（把下面的路径换成你真实存在的文件）
REM     例如：decoder 用 coco_cos.pt，一张干净图和一张目标图
python demo.py ^
  --decoder_path "E:\work\adv_attack\checkpoints\coco_cos.pt" ^
  --clean_image_path "D:\your_data\clean.jpg" ^
  --target_image_path "D:\your_data\target.jpg" ^
  --output_path "E:\work\adv_attack\output.png"

REM 11. 启动本地 UI
streamlit run ui\app.py
